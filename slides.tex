\begin{frame}{Colors}
    The theme mainly relies on two colors, the primary color \tikz{\fill[fill=\theprimarycolor] (0,0) rectangle ++(10pt,10pt)}
    and the secondary color \tikz{\fill[fill=\thesecondarycolor] (0,0) rectangle ++(10pt,10pt)}. For example, the secondary color 
    is used to \highlight{highlight text}.\\[2em]

\end{frame}


{
    \primarycolor{red}
    \secondarycolor{green}
    \begin{frame}[fragile]{Colors}
        The theme mainly relies on two colors, the primary color \tikz{\fill[fill=\theprimarycolor] (0,0) rectangle ++(10pt,10pt)}
        and the secondary color \tikz{\fill[fill=\thesecondarycolor] (0,0) rectangle ++(10pt,10pt)}. For example, the secondary color 
        is used to \highlight{highlight text}.\\[2em]

        To change the primary color to \texttt{red} and the secondary color to \verb|green|,
        use \verb|\primarycolor{red}| and \verb|\secondarycolor{green}|. 
    \end{frame}
}

\begin{frame}[fragile]{Highlighting}
    You can highlight \highlight{text} with \verb|\highlight{text}|. \verb|\highlight| is compatible 
    with beamer overlays. Thus, to only highlight \highlight<2>{text} on the second slide you can 
    simply write \verb|\highlight<2>{text}|.\\[2em]


    To change the highlighting color of the \highlight[red]{text}, pass the color 
    as an optional argument \verb|\highlight[red]{text}|.
\end{frame}

\begin{frame}[fragile]{Frame Numbers}
    The color of the frame numbers can be controlled with 
    \verb|\framenumbercolor{red}|. The background with
    \verb|\framenumberbackground{...}|.
\end{frame}

{
    \framenumbercolor{\thesecondarycolor}
    \framenumberbackground
    {{\color{\theprimarycolor}\huge$\blacksquare$}}
\begin{frame}[fragile]{Frame Numbers}
    The color of the frame numbers can be controlled with 
    \verb|\framenumbercolor|. The background with
    \verb|\framenumberbackground|.\vspace{2em}

    For example, we can replace the background with a heart and 
    make the counter white:
    \begin{verbatim}
\framenumbercolor{\thesecondarycolor}
\framenumberbackground
    {{\color{\theprimarycolor}\huge$\blacksquare$}}
    \end{verbatim}
\end{frame}
}

\begin{frame}{Timeline}
    
    \begin{textblock*}{\paperwidth} (0cm,5pt)
        \begin{center}
            \begin{tikzpicture}
                \coordinate (top) at (0,0);
                \coordinate[below=8 of top] (bottom);
                \coordinate[below=2 of top] (one);
                \coordinate[below=3.5 of top] (two);
                \coordinate[below=5 of top] (three);
                \coordinate[below=6.5 of top] (four);
    
                \draw[->, \theprimarycolor] (top) -- (bottom);
    
                \visible<2->{
                    \begin{timelinenode}[right]{one}{.4\linewidth}{First Item}
                        Text here.
                    \end{timelinenode}
                }
                \visible<3->{
                    \begin{timelinenode}[left]{two}{.4\linewidth}{Second Item}
                        Text here.
                    \end{timelinenode}
                }
                \visible<4->{
                    \begin{timelinenode}[right]{three}{.4\linewidth}{Third Item}
                        Text here.
                    \end{timelinenode}
                }
                \visible<5->{
                    \begin{graytimelinenode}[left]{four}{.4\linewidth}{Fourth Item}
                        Text here.
                    \end{graytimelinenode}
                }
            \end{tikzpicture}        
        \end{center}
    \end{textblock*}
\end{frame}
